## 管理文件

```kotlin
// 列出文件夹内容
fun listFolder(documentFile: DocumentFile): List<DocumentFile> {
    return documentFile.listFiles().toList()
}

// 创建文件
fun createFile(parent: DocumentFile, name: String, mimeType: String): DocumentFile? {
    return parent.createFile(mimeType, name)
}

// 删除
fun delete(documentFile: DocumentFile): Boolean {
    return documentFile.delete()
}
```
