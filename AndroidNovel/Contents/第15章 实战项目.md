## 2. API 接口

“然后定义 API 接口。”小 Com 继续：

```kotlin
interface WeatherApi {
    @GET("weather")
    suspend fun getCurrentWeather(
        @Query("city") city: String,
        @Query("key") key: String
    ): CurrentWeatherResponse
    
    @GET("forecast")
    suspend fun getForecast(
        @Query("city") city: String,
        @Query("key") key: String
    ): ForecastResponse
}

data class CurrentWeatherResponse(
    val code: Int,
    val data: CurrentWeather
)

data class ForecastResponse(
    val code: Int,
    val data: List<Forecast>
)
```

## 3. Repository

“然后是 Repository——数据仓库。”小 Com 展示了：

```kotlin
class WeatherRepository(
    private val api: WeatherApi,
    private val cityDao: CityDao
) {
    // 获取天气（带缓存）
    suspend fun getWeather(city: String): Result<CurrentWeather> {
        return try {
            val response = api.getCurrentWeather(city, API_KEY)
            if (response.code == 200) {
                Result.success(response.data)
            } else {
                Result.failure(Exception("获取天气失败"))
            }
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
    
    // 获取预报
    suspend fun getForecast(city: String): Result<List<Forecast>> {
        return try {
            val response = api.getForecast(city, API_KEY)
            if (response.code == 200) {
                Result.success(response.data)
            } else {
                Result.failure(Exception("获取预报失败"))
            }
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
    
    // 收藏城市
    fun getFavoriteCities(): Flow<List<City>> = cityDao.getAllCitiesFlow()
    
    suspend fun addFavoriteCity(city: City) {
        cityDao.insertCity(city)
    }
    
    suspend fun removeFavoriteCity(city: City) {
        cityDao.deleteCity(city)
    }
    
    companion object {
        // 这里放免费的天气API Key
        private const val API_KEY = "your_api_key"
    }
}
```

## 4. ViewModel

“然后是 ViewModel——业务逻辑。”小 Com 展示了：

```kotlin
@HiltViewModel
class WeatherViewModel(
    private val repository: WeatherRepository
) : ViewModel() {
    
    // 当前天气状态
    private val _currentWeather = MutableStateFlow<UiState<CurrentWeather>>(UiState.Initial)
    val currentWeather: StateFlow<UiState<CurrentWeather>> = _currentWeather
    
    // 预报状态
    private val _forecast = MutableStateFlow<UiState<List<Forecast>>>(UiState.Initial)
    val forecast: StateFlow<UiState<List<Forecast>>> = _forecast
    
    // 搜索城市
    private val _searchQuery = MutableStateFlow("")
    val searchQuery: StateFlow<String> = _searchQuery
    
    // 收藏城市
    val favoriteCities: StateFlow<List<City>> = repository.getFavoriteCities()
        .stateIn(viewModelScope, SharingStarted.Lazily, emptyList())
    
    // 加载天气
    fun loadWeather(city: String) {
        viewModelScope.launch {
            _currentWeather.value = UiState.Loading
            
            repository.getWeather(city)
                .onSuccess { weather ->
                    _currentWeather.value = UiState.Success(weather)
                }
                .onFailure { e ->
                    _currentWeather.value = UiState.Error(e.message ?: "加载失败")
                }
        }
        
        viewModelScope.launch {
            _forecast.value = UiState.Loading
            
            repository.getForecast(city)
                .onSuccess { list ->
                    _forecast.value = UiState.Success(list)
                }
                .onFailure { e ->
                    _forecast.value = UiState.Error(e.message ?: "加载失败")
                }
        }
    }
    
    // 更新搜索
    fun updateSearchQuery(query: String) {
        _searchQuery.value = query
    }
    
    // 搜索城市
    fun searchCity() {
        val query = _searchQuery.value
        if (query.isNotBlank()) {
            loadWeather(query)
        }
    }
    
    // 收藏城市
    fun toggleFavorite(city: City) {
        viewModelScope.launch {
            if (city.isFavorite) {
                repository.removeFavoriteCity(city)
            } else {
                repository.addFavoriteCity(city.copy(isFavorite = true))
            }
        }
    }
}

sealed class UiState<out T> {
    object Initial : UiState()
    object Loading : UiState()
    data class Success<T>(val data: T) : UiState()
    data class Error(val message: String) : UiState()
}
```

## 5. UI - 主界面

“然后是 UI—— Compose 界面。”小 Com 展示了主界面：

```kotlin
@Composable
fun WeatherScreen(
    viewModel: WeatherViewModel = viewModel()
) {
    val currentWeather by viewModel.currentWeather.collectAsState()
    val forecast by viewModel.forecast.collectAsState()
    val searchQuery by viewModel.searchQuery.collectAsState()
    
    Column(modifier = Modifier.fillMaxSize()) {
        // 搜索栏
        SearchBar(
            query = searchQuery,
            onQueryChange = viewModel::updateSearchQuery,
            onSearch = viewModel::searchCity,
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp)
        )
        
        // 内容区
        when (val state = currentWeather) {
            is UiState.Initial -> {
                Box(
                    modifier = Modifier.fillMaxSize(),
                    contentAlignment = Alignment.Center
                ) {
                    Text("搜索城市查看天气")
                }
            }
            is UiState.Loading -> {
                Box(
                    modifier = Modifier.fillMaxSize(),
                    contentAlignment = Alignment.Center
                ) {
                    CircularProgressIndicator()
                }
            }
            is UiState.Success -> {
                WeatherContent(
                    weather = state.data,
                    forecast = (forecast as? UiState.Success)?.data ?: emptyList(),
                    onFavoriteClick = { viewModel.toggleFavorite(it) }
                )
            }
            is UiState.Error -> {
                Box(
                    modifier = Modifier.fillMaxSize(),
                    contentAlignment = Alignment.Center
                ) {
                    Column(horizontalAlignment = Alignment.CenterHorizontally) {
                        Text(state.message, color = Color.Red)
                        Spacer(modifier = Modifier.height(8.dp))
                        Button(onClick = { viewModel.searchCity() }) {
                            Text("重试")
                        }
                    }
                }
            }
        }
    }
}
```

@Composable
fun WeatherContent(
    weather: CurrentWeather,
    forecast: List<Forecast>,
    onFavoriteClick: (City) -> Unit
) {
    LazyColumn(
        modifier = Modifier.fillMaxSize(),
        contentPadding = PaddingValues(16.dp)
    ) {
        // 当前天气卡片
        item {
            CurrentWeatherCard(weather = weather)
            Spacer(modifier = Modifier.height(24.dp))
        }
        
        // 天气预报标题
        item {
            Text(
                text = "未来预报",
                style = MaterialTheme.typography.titleLarge
            )
            Spacer(modifier = Modifier.height(12.dp))
        }
        
        // 预报列表
        items(forecast) { day ->
            ForecastItem(forecast = day)
            Spacer(modifier = Modifier.height(8.dp))
        }
    }
}

@Composable
fun CurrentWeatherCard(weather: CurrentWeather) {
    Card(
        modifier = Modifier.fillMaxWidth(),
        colors = CardDefaults.cardColors(
            containerColor = Color(0xFF2196F3)
        ),
        shape = RoundedCornerShape(24.dp)
    ) {
        Column(
            modifier = Modifier.padding(24.dp),
            horizontalAlignment = Alignment.CenterHorizontally
        ) {
            Text(
                text = weather.city,
                style = MaterialTheme.typography.headlineMedium,
                color = Color.White
            )
            
            Spacer(modifier = Modifier.height(16.dp))
            
            Text(
                text = "${weather.temp}°",
                style = MaterialTheme.typography.displayLarge,
                color = Color.White
            )
            
            Text(
                text = weather.condition,
                style = MaterialTheme.typography.titleMedium,
                color = Color.White.copy(alpha = 0.8f)
            )
            
            Spacer(modifier = Modifier.height(24.dp))
            
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceEvenly
            ) {
                WeatherDetail(icon = Icons.Default.Water, value = "${weather.humidity}%")
                WeatherDetail(icon = Icons.Default.Air, value = "${weather.windSpeed} m/s")
            }
        }
    }
}

@Composable
fun WeatherDetail(icon: ImageVector, value: String) {
    Row(verticalAlignment = Alignment.CenterVertically) {
        Icon(
            icon,
            contentDescription = null,
            tint = Color.White.copy(alpha = 0.8f)
        )
        Spacer(modifier = Modifier.width(4.dp))
        Text(
            text = value,
            color = Color.White
        )
    }
}

## 7. 搜索栏

“搜索栏组件：”

```kotlin
@Composable
fun SearchBar(
    query: String,
    onQueryChange: (String) -> Unit,
    onSearch: () -> Unit,
    modifier: Modifier = Modifier
) {
    OutlinedTextField(
        value = query,
        onValueChange = onQueryChange,
        modifier = modifier,
        placeholder = { Text("搜索城市") },
        leadingIcon = {
            Icon(Icons.Default.Search, contentDescription = null)
        },
        trailingIcon = {
            if (query.isNotEmpty()) {
                IconButton(onClick = onSearch) {
                    Icon(Icons.Default.Send, contentDescription = "搜索")
                }
            }
        },
        singleLine = true,
        shape = RoundedCornerShape(24.dp),
        keyboardOptions = KeyboardOptions(imeAction = ImeAction.Search),
        keyboardActions = KeyboardActions(onSearch = { onSearch() })
    )
}
```
