## 2. 创建通知渠道（Android 8+）

“Android 8+ 需要创建通知渠道。”小 Com 展示了：

```kotlin
class App : Application() {
    
    override fun onCreate() {
        super.onCreate()
        
        // 创建通知渠道
        val channel = NotificationChannel(
            CHANNEL_ID,           // 渠道 ID
            "重要通知",             // 渠道名称
            NotificationManager.IMPORTANCE_HIGH  // 重要性
        ).apply {
            description = "用于接收重要消息推送"  // 描述
            enableVibration(true)  // 振动
            enableLights(true)     // 呼吸灯
        }
        
        val notificationManager = getSystemService(NotificationManager::class.java)
        notificationManager.createNotificationChannel(channel)
    }
    
    companion object {
        const val CHANNEL_ID = "important_channel"
    }
}
```

## 3. 发送简单通知

“先看最简单的通知。”小 Com 展示了：

```kotlin
fun sendSimpleNotification(context: Context) {
    val notification = NotificationCompat.Builder(context, App.CHANNEL_ID)
        .setSmallIcon(R.drawable.ic_notification)       // 小图标（必填）
        .setContentTitle("标题")                          // 标题
        .setContentText("内容")                          // 内容
        .setPriority(NotificationCompat.PRIORITY_HIGH)  // 优先级
        .setAutoCancel(true)                            // 点击后消失
        .build()
    
    val notificationManager = context.getSystemService(NotificationManager::class.java)
    notificationManager.notify(1, notification)
}
```

fun sendRichNotification(context: Context) {

        .setPriority(NotificationCompat.PRIORITY_HIGH)
        .setAutoCancel(true)
        .build()
    
    val notificationManager = context.getSystemService(NotificationManager::class.java)
    notificationManager.notify(2, notification)
}

## 5. 带操作按钮的通知

“可以添加操作按钮。”小 Com 展示了：

```kotlin
fun sendNotificationWithActions(context: Context) {
    // 创建 PendingIntent
    val intent = Intent(context, MainActivity::class.java).apply {
        flags = Intent.FLAG_ACTIVITY_NEW_TASK or Intent.FLAG_ACTIVITY_CLEAR_TASK
    }
    val pendingIntent = PendingIntent.getActivity(
        context, 0, intent, PendingIntent.FLAG_IMMUTABLE
    )
    
    // 回复操作的 PendingIntent
    val replyIntent = Intent(context, ReplyReceiver::class.java)
    val replyPendingIntent = PendingIntent.getBroadcast(
        context, 0, replyIntent, PendingIntent.FLAG_IMMUTABLE
    )
    
    val notification = NotificationCompat.Builder(context, App.CHANNEL_ID)
        .setSmallIcon(R.drawable.ic_notification)
        .setContentTitle("新消息")
        .setContentText("点击查看详情")
        
        // 添加操作按钮
        .addAction(R.drawable.ic_reply, "回复", replyPendingIntent)
        .addAction(R.drawable.ic_delete, "删除", pendingIntent)
        
        // 设置点击行为
        .setContentIntent(pendingIntent)
        .setAutoCancel(true)
        .build()
    
    val notificationManager = context.getSystemService(NotificationManager::class.java)
    notificationManager.notify(3, notification)
}
```

## 9. 更新和取消通知

“可以更新和取消通知。”小 Com 展示了：

```kotlin
fun updateNotification(context: Context) {
    val notification = NotificationCompat.Builder(context, App.CHANNEL_ID)
        .setSmallIcon(R.drawable.ic_notification)
        .setContentTitle("标题已更新")
        .setContentText("内容也更新了")
        .build()
    
    // 使用相同的 ID 来更新
    val notificationManager = context.getSystemService(NotificationManager::class.java)
    notificationManager.notify(1, notification)
}

fun cancelNotification(context: Context) {
    val notificationManager = context.getSystemService(NotificationManager::class.java)
    notificationManager.cancel(1)  // 取消指定通知
    
    // 或者取消所有
    notificationManager.cancelAll()
}
```

## 10. 实战：完整的通知工具类

“我们来做最后一个练习——完整的通知工具类！”小 Com 提议道。

```kotlin
object NotificationHelper {
    
    private const val CHANNEL_ID = "app_channel"
    private const val CHANNEL_NAME = "应用通知"
    
    fun createChannel(context: Context) {
        val channel = NotificationChannel(
            CHANNEL_ID,
            CHANNEL_NAME,
            NotificationManager.IMPORTANCE_DEFAULT
        ).apply {
            description = "应用普通通知"
            enableVibration(true)
        }
        
        val manager = context.getSystemService(NotificationManager::class.java)
        manager.createNotificationChannel(channel)
    }
    
    fun showSimpleNotification(
        context: Context,
        title: String,
        message: String
    ) {
        val notification = NotificationCompat.Builder(context, CHANNEL_ID)
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle(title)
            .setContentText(message)
            .setPriority(NotificationCompat.PRIORITY_DEFAULT)
            .setAutoCancel(true)
            .build()
        
        context.getSystemService(NotificationManager::class.java)
            .notify(System.currentTimeMillis().toInt(), notification)
    }
    
    fun showNotificationWithIntent(
        context: Context,
        title: String,
        message: String,
        activityClass: Class<*>
    ) {
        val intent = Intent(context, activityClass).apply {
            flags = Intent.FLAG_ACTIVITY_NEW_TASK or Intent.FLAG_ACTIVITY_CLEAR_TASK
        }
        
        val pendingIntent = PendingIntent.getActivity(
            context,
            0,
            intent,
            PendingIntent.FLAG_IMMUTABLE
        )
        
        val notification = NotificationCompat.Builder(context, CHANNEL_ID)
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle(title)
            .setContentText(message)
            .setContentIntent(pendingIntent)
            .setAutoCancel(true)
            .build()
        
        context.getSystemService(NotificationManager::class.java)
            .notify(System.currentTimeMillis().toInt(), notification)
    }
}

// 使用
@Composable
fun NotificationScreen() {
    val context = LocalContext.current
    
    LaunchedEffect(Unit) {
        NotificationHelper.createChannel(context)
    }
    
    Column {
        Button(onClick = {
            NotificationHelper.showSimpleNotification(
                context,
                "你好",
                "这是一条通知"
            )
        }) {
            Text("发送通知")
        }
        
        Button(onClick = {
            NotificationHelper.showNotificationWithIntent(
                context,
                "点击打开",
                "点击跳转到详情",
                DetailActivity::class.java
            )
        }) {
            Text("发送可点击的通知")
        }
    }
}
```
