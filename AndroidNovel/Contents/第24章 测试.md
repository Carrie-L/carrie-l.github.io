## 3. 单元测试：ViewModel

“先看 ViewModel 的单元测试。”小 Com 展示了：

```kotlin
// 待测试的 ViewModel
@HiltViewModel
class CounterViewModel @Inject constructor() : ViewModel() {
    
    private val _count = MutableStateFlow(0)
    val count: StateFlow<Int> = _count
    
    fun increment() {
        _count.value++
    }
    
    fun decrement() {
        _count.value--
    }
}

// 测试代码
class CounterViewModelTest {
    
    private lateinit var viewModel: CounterViewModel
    
    @Before
    fun setup() {
        viewModel = CounterViewModel()
    }
    
    @Test
    fun `初始值应该是 0`() = runTest {
        assertEquals(0, viewModel.count.value)
    }
    
    @Test
    fun `increment 应该增加计数`() = runTest {
        viewModel.increment()
        assertEquals(1, viewModel.count.value)
        
        viewModel.increment()
        assertEquals(2, viewModel.count.value)
    }
    
    @Test
    fun `decrement 应该减少计数`() = runTest {
        viewModel.increment()
        viewModel.increment()
        assertEquals(2, viewModel.count.value)
        
        viewModel.decrement()
        assertEquals(1, viewModel.count.value)
    }
}
```

## 4. 单元测试：Repository

“然后测试 Repository。”小 Com 展示了：

```kotlin
// 待测试的 Repository
class UserRepository(
    private val api: ApiService,
    private val userDao: UserDao
) {
    suspend fun getUser(id: String): User {
        // 先尝试从本地获取
        val localUser = userDao.getUserById(id)
        if (localUser != null) return localUser
        
        // 本地没有，从网络获取
        val remoteUser = api.getUser(id)
        userDao.insertUser(remoteUser)
        return remoteUser
    }
}

// 测试代码
class UserRepositoryTest {
    
    private val api = mockk<ApiService>()
    private val userDao = mockk<UserDaoDao>(relaxed = true)
    private val repository = UserRepository(api, userDao)
    
    @Test
    fun `从本地获取用户`() = runTest {
        // Arrange
        val localUser = User("1", "本地用户", "local@test.com")
        coEvery { userDao.getUserById("1") } returns localUser
        
        // Act
        val result = repository.getUser("1")
        
        // Assert
        assertEquals("本地用户", result.name)
        coVerify(exactly = 0) { api.getUser("1") }  // 没有调用 API
    }
    
    @Test
    fun `本地没有时从网络获取`() = runTest {
        // Arrange
        val remoteUser = User("2", "远程用户", "remote@test.com")
        coEvery { userDao.getUserById("2") } returns null
        coEvery { api.getUser("2") } returns remoteUser
        
        // Act
        val result = repository.getUser("2")
        
        // Assert
        assertEquals("远程用户", result.name)
        coVerify { api.getUser("2") }
        coVerify { userDao.insertUser(remoteUser) }
    }
}
```

## 5. UI 测试：Compose

```kotlin
// 待测试的界面
@Composable
fun LoginScreen(
    onLoginClick: (String, String) -> Unit
) {
    var username by remember { mutableStateOf("") }
    var password by remember { mutableStateOf("") }
    
    Column(
        modifier = Modifier.fillMaxSize(),
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        OutlinedTextField(
            value = username,
            onValueChange = { username = it },
            label = { Text("用户名") },
            modifier = Modifier.testTag("username")
        )
        
        OutlinedTextField(
            value = password,
            onValueChange = { password = it },
            label = { Text("密码") },
            modifier = Modifier.testTag("password")
        )
        
        Button(
            onClick = { onLoginClick(username, password) },
            modifier = Modifier.testTag("login_button"),
            enabled = username.isNotEmpty() && password.isNotEmpty()
        ) {
            Text("登录")
        }
    }
}
```

## 6. 编写 UI 测试

“编写 UI 测试。”小 Com 展示了：

```kotlin
class LoginScreenTest {
    
    @get:Rule
    val composeTestRule = createComposeRule()
    
    @Test
    fun `登录按钮初始应该禁用`() {
        composeTestRule.setContent {
            LoginScreen(onLoginClick = { _, _ -> })
        }
        
        // 检查按钮是否禁用
        composeTestRule.onNodeWithTag("login_button")
            .assertIsDisabled()
    }
    
    @Test
    fun `输入用户名和密码后按钮应该启用`() {
        composeTestRule.setContent {
            LoginScreen(onLoginClick = { _, _ -> })
        }
        
        // 输入用户名
        composeTestRule.onNodeWithTag("username")
            .performTextInput("testuser")
        
        // 输入密码
        composeTestRule.onNodeWithTag("password")
            .performTextInput("password123")
        
        // 检查按钮是否启用
        composeTestRule.onNodeWithTag("login_button")
            .assertIsEnabled()
    }
    
    @Test
    fun `点击登录按钮应该调用回调`() {
        var loginCalled = false
        
        composeTestRule.setContent {
            LoginScreen { username, password ->
                loginCalled = true
                assertEquals("testuser", username)
                assertEquals("password123", password)
            }
        }
        
        // 输入内容
        composeTestRule.onNodeWithTag("username")
            .performTextInput("testuser")
        composeTestRule.onNodeWithTag("password")
            .performTextInput("password123")
        
        // 点击按钮
        composeTestRule.onNodeWithTag("login_button")
            .performClick()
        
        // 验证回调被调用
        assertTrue(loginCalled)
    }
}
```

## 8. 实战：完整测试

“我们来做最后一个练习——完整的测试！”小 Com 提议道。

```kotlin
// 1. 测试 ViewModel
@HiltViewModel
class UserListViewModel @Inject constructor(
    private val repository: UserRepository
) : ViewModel() {
    
    private val _users = MutableStateFlow<UiState<List<User>>>(UiState.Loading)
    val users: StateFlow<UiState<List<User>>> = _users
    
    fun loadUsers() {
        viewModelScope.launch {
            _users.value = UiState.Loading
            try {
                val userList = repository.getUsers()
                _users.value = UiState.Success(userList)
            } catch (e: Exception) {
                _users.value = UiState.Error(e.message ?: "加载失败")
            }
        }
    }
}

// 2. ViewModel 测试
class UserListViewModelTest {
    
    @Mock
    private lateinit var repository: UserRepository
    
    private lateinit var viewModel: UserListViewModel
    
    @Before
    fun setup() {
        viewModel = UserListViewModel(repository)
    }
    
    @Test
    fun `加载用户应该成功`() = runTest {
        // Arrange
        val mockUsers = listOf(User("1", "张三"), User("2", "李四"))
        coEvery { repository.getUsers() } returns mockUsers
        
        // Act
        viewModel.loadUsers()
        
        // Assert
        val state = viewModel.users.value
        assertTrue(state is UiState.Success)
        assertEquals(2, (state as UiState.Success).data.size)
    }
    
    @Test
    fun `加载失败应该显示错误`() = runTest {
        // Arrange
        coEvery { repository.getUsers() } throws Exception("网络错误")
        
        // Act
        viewModel.loadUsers()
        
        // Assert
        val state = viewModel.users.value
        assertTrue(state is UiState.Error)
    }
}

// 3. UI 测试
class UserListScreenTest {
    
    @get:Rule
    val composeTestRule = createComposeRule()
    
    @Test
    fun `加载中应该显示进度条`() {
        composeTestRule.setContent {
            val viewModel = mockk<UserListViewModel>(relaxed = true)
            
            // Mock 加载状态
            every { viewModel.users } returns MutableStateFlow(UiState.Loading)
            
            UserListScreen(viewModel = viewModel)
        }
        
        composeTestRule.onNodeWithText("加载中...")
            .assertIsDisplayed()
    }
}
```

## 测试最佳实践

小 Com 总结了测试最佳实践：

| 原则 | 说明 |
|------|------|
| **AAA** | Arrange-Act-Assert |
| **命名** | 测试名要描述行为 |
| **独立** | 测试之间不要相互依赖 |
| **覆盖** | 关键路径都要测试 |
| **快速** | 单元测试要快 |
