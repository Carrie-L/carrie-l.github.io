## 2. 对称加密

“先看对称加密。”小 Com 展示了：

**AES 是最常用的对称加密：**

```kotlin
// AES 加密
fun encryptAES(data: String, key: ByteArray): ByteArray {
    val cipher = Cipher.getInstance("AES")
    val secretKey = SecretKeySpec(key, "AES")
    cipher.init(Cipher.ENCRYPT_MODE, secretKey)
    return cipher.doFinal(data.toByteArray())
}

// AES 解密
fun decryptAES(data: ByteArray, key: ByteArray): String {
    val cipher = Cipher.getInstance("AES")
    val secretKey = SecretKeySpec(key, "AES")
    cipher.init(Cipher.DECRYPT_MODE, secretKey)
    return String(cipher.doFinal(data))
}
```

fun generateKeyPair(): KeyPair {

// RSA 加密

    return cipher.doFinal(data)

// RSA 解密
fun decryptRSA(data: ByteArray, privateKey: PrivateKey): ByteArray {
    val cipher = Cipher.getInstance("RSA")

## 4. 哈希

“哈希用于验证数据完整性。”小 Com 展示了：

```kotlin
// MD5（不安全）
fun md5(input: String): String {
    val md = MessageDigest.getInstance("MD5")
    return md.digest(input.toByteArray()).joinToString("") { "%02x".format(it) }
}

// SHA-256（安全）
fun sha256(input: String): String {
    val sha = MessageDigest.getInstance("SHA-256")
    return sha.digest(input.toByteArray()).joinToString("") { "%02x".format(it) }
}

// 使用场景
fun checkPassword(input: String, stored: String): Boolean {
    return sha256(input) == stored
}
```

## 5. 数字签名

“数字签名用于验证来源。”小 Com 展示了：

```kotlin
// 生成签名
fun sign(data: ByteArray, privateKey: PrivateKey): ByteArray {
    val signature = Signature.getInstance("SHA256withRSA")
    signature.initSign(privateKey)
    signature.update(data)
    return signature.sign()
}

// 验证签名
fun verify(data: ByteArray, signature: ByteArray, publicKey: PublicKey): Boolean {
    val sig = Signature.getInstance("SHA256withRSA")
    sig.initVerify(publicKey)
    sig.update(data)
    return sig.verify(signature)
}
```

## 6. Android Keystore

“用 Keystore 保护密钥。”小 Com 展示了：

```kotlin
// 生成密钥到 Keystore
fun generateKeyInKeystore(alias: String): PrivateKey {
    val keyGenerator = KeyGenerator.getInstance(
        KeyProperties.KEY_ALGORITHM_AES,
        "AndroidKeyStore"
    )
    
    val spec = KeyGenParameterSpec.Builder(
        alias,
        KeyProperties.PURPOSE_ENCRYPT or KeyProperties.PURPOSE_DECRYPT
    )
        .setBlockModes(KeyProperties.BLOCK_MODE_GCM)
        .setEncryptionPaddings(KeyProperties.ENCRYPTION_PADDING_NONE)
        .setKeySize(256)
        .build()
    
    keyGenerator.init(spec)
    return keyGenerator.generateKey()
}

// 使用 Keystore 密钥
fun encryptWithKeystore(data: ByteArray, alias: String): ByteArray {
    val keyStore = KeyStore.getInstance("AndroidKeyStore")
    keyStore.load(null)
    val key = keyStore.getKey(alias, null) as Key
    
    val cipher = Cipher.getInstance("AES/GCM/NoPadding")
    cipher.init(Cipher.ENCRYPT_MODE, key)
    return cipher.doFinal(data)
}
```

## 7. 加固 App

**2. 完整性校验**

```kotlin
fun checkSignature(context: Context): Boolean {
    val packageInfo = context.packageManager.getPackageInfo(
        context.packageName,
        PackageManager.GET_SIGNATURES
    )
    val signature = packageInfo.signatures[0].toByteArray()
    val md = MessageDigest.getInstance("SHA")
    val digest = md.digest(signature)
    return digest.contentEquals(EXPECTED_SIGNATURE)
}
```

**3. 防止调试**

```kotlin
fun isDebuggable(context: Context): Boolean {
    return (context.applicationInfo.flags and ApplicationInfo.FLAG_DEBUGGABLE) != 0
}

fun antiDebug() {
    if (isDebuggable(Application())) {
        android.os.Process.killProcess(android.os.Process.myPid())
    }
}
```

## 8. 安全存储

// 使用 EncryptedSharedPreferences
fun getSecurePrefs(context: Context): SharedPreferences {
    val masterKey = MasterKey.Builder(context)
        .setKeyScheme(MasterKey.KeyScheme.AES256_GCM)
        .build()
    
    return EncryptedSharedPreferences.create(
        context,
        "secure_prefs",
        masterKey,
        EncryptedSharedPreferences.PrefKeyEncryptionScheme.AES256_SIV,
        EncryptedSharedPreferences.PrefValueEncryptionScheme.AES256_GCM
    )
}

class SecurityManager(private val context: Context) {
    
    // 1. 检查签名
    fun verifySignature(): Boolean {
        // 检查签名是否匹配
    }
    
    // 2. 检查调试
    fun detectDebugger(): Boolean {
        return Debug.isDebuggerConnected()
    }

    fun secureSave(key: String, value: String) {
        val prefs = getSecurePrefs(context)
        prefs.edit().putString(key, value).apply()
    }
    
    // 4. 安全网络
    fun createSecureRetrofit(): Retrofit {
        return Retrofit.Builder()
            .baseUrl("https://api.example.com/")
            .client(createSecureClient())
            .build()
    }
    
    // 5. 初始化
    fun init() {
        if (!verifySignature()) {
            throw SecurityException("签名验证失败")
        }
        if (detectDebugger()) {
            android.os.Process.killProcess(android.os.Process.myPid())
        }
    }
}
